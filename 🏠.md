function deveIncluir(pasta) {

    const upperFolder = folder.toUpperCase();
    for (let key in ICONES) {
        if (upperFolder.includes(key)) return ICONES[key];

    return ICONES.DEFAULT;

    const diff = Date.now() - data.toJSDate().getTime();
